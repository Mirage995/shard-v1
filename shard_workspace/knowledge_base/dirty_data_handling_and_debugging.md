## Practical Example
```python
# Data Cleaning Example (Python)
import re

def clean_phone_number(phone_number):
    """Removes non-numeric characters from a phone number."""
    return re.sub(r'\D', '', phone_number)

dirty_number = "(123) 456-7890"
clean_number = clean_phone_number(dirty_number)
print(f"Cleaned number: {clean_number}")
```
