## Practical Example
```python
import OpenGL
from OpenGL.GL import *
from OpenGL.GL.shaders import compileProgram, compileShader

# Create a vertex array object (VAO) and a vertex buffer object (VBO)
vao = glGenVertexArrays(1)
vbo = glGenBuffers(1)

# Bind the VAO and VBO, and specify the vertex data
glBindVertexArray(vao)
glBindBuffer(GL_ARRAY_BUFFER, vbo)
glBufferData(GL_ARRAY_BUFFER, numpy.array([-0.5, -0.5, 0.0, 0.5, -0.5, 0.0, 0.0, 0.5, 0.0], dtype=numpy.float32), GL_STATIC_DRAW)

# Specify the vertex attributes and draw the vertices
glVertexAttribPointer(0, 3, GL_FLOAT, False, 0, None)
glEnableVertexAttribArray(0)
glDrawArrays(GL_TRIANGLES, 0, 3)
```
