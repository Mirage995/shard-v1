## Practical Example
```python
from bs4 import BeautifulSoup

class InvalidHTMLError(Exception):
    """Raised when HTML is invalid."""
    pass

def parse_html(html_content):
    """Parses HTML content and raises a custom exception if invalid."""
    try:
        soup = BeautifulSoup(html_content, 'html.parser')
        if not soup.find('body'):
            raise InvalidHTMLError("HTML document must contain a <body> tag.")
        return soup
    except Exception as e:
        raise InvalidHTMLError(f"Failed to parse HTML: {e}")

# Example usage
html_string = "<html><head><title>Test</title></head></html>" # Missing body
try:
    parsed_html = parse_html(html_string)
    print("HTML parsed successfully.")
except InvalidHTMLError as e:
    print(f"Error: {e}")
```
