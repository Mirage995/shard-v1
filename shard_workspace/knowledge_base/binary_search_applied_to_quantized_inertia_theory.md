## Key Concepts

- **Parameter Space Search**: Using binary search to efficiently explore QI parameter ranges (thrust predictions, acceleration thresholds, cosmic acceleration values)

def find_qi_threshold(acceleration_data, classical_predictions, qi_predictions):
    """
    Binary search to find acceleration threshold where QI diverges from classical mechanics.
    Assumes sorted acceleration_data in ascending order.
    """
    left, right = 0, len(acceleration_data) - 1
    threshold_index = -1
    
    while left <= right:
        mid = (left + right) // 2
        
        # Calculate relative difference between predictions
        classical = classical_predictions[mid]
        qi = qi_predictions[mid]
        divergence = abs(qi - classical) / classical
        
        # Check if divergence exceeds 10% threshold
        if divergence > 0.10:
            threshold_index = mid
            right = mid - 1  # Search for earlier threshold
        else:
            left = mid + 1
    
    if threshold_index != -1:
        return acceleration_data[threshold_index]
    return None

# Example usage: Find where QI effects become significant
accelerations = [1e-12, 1e-11, 1e-10, 1e-9, 1e-8]  # m/s² (sorted)
classical = [0.98, 0.95, 0.90, 0.80, 0.65]  # normalized predictions
qi_model = [0.99, 0.96, 0.85, 0.60, 0.35]  # QI predictions

critical_a = find_qi_threshold(accelerations, classical, qi_model)
print(f"QI effects significant at a ≈ {critical_a} m/s²")
