import numpy as np

# Simplified example of active inference for navigation
def active_inference(state, action, observation):
    # Compute the probability of the observation given the state and action
    prob_obs = np.random.rand()
    # Update the state based on the observation and action
    new_state = state + action + np.random.randn()
    return new_state, prob_obs

# Initialize the state and action
state = 0.0
action = 1.0

# Perform active inference for 10 steps
for _ in range(10):
    state, prob_obs = active_inference(state, action, np.random.rand())
    print(f"State: {state}, Probability of Observation: {prob_obs}")
